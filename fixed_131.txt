1) correctly pass selected language to setNewLangauge (rg2getjson.js)
getScript.done call returns script content into first variable so it can't be used to pass lang variable for sub routine.
Here setNewLanguage was every time called with script content and not with actual language code like 'de'
Even though setNewLanguage worked as lang variable was not actually used anything else than check against 'en'

  function getNewLanguage(lang) {
    $.getScript(rg2Config.lang_url + lang + ".js")
      .done(function () {
        // script sets rg2.dictionary to new language
        rg2.ui.setNewLanguage(lang);
      }).fail(function (jqxhr, settings, exception) {
        /*jslint unparam:true*/
        /* jshint unused:vars */
        reportJSONFail("Language request failed.");
      });
  }

2) select correct "start language" for language dropdown menu (config.js )
Options language drop down menu is created just before dictionary.code is set by dynamically loaded xx.js language script in getNewLanguage/setNewLanguage.
In a case of non english rg2.config.start_langauge the drop down menu was not initially showing any language as selected.

  function setLanguageOptions() {
    // use English until we load something else
    dictionary = {};
    dictionary.code = 'en';
    if (rg2Config.start_language !== "en") {
      rg2.getNewLanguage(rg2Config.start_language);
      // temporarily set start language code
      dictionary.code = rg2Config.start_language;
    }
    // set available languages and set start language if requested
    rg2.createLanguageDropdown(rg2.config.languages);
  }

3) rg2.events used before initialization (rg2ui.js)
On init function language files are loaded dynamically from xx.js by rg2.setLanguageOptions and its sub routine getNewLanguage/setNewLanguage which also calls CreateEventMenu and rg2.events in it.
init()
  ->rg2.setLanguageOptions(rg2.js) ->setLanguageOptions (config.js) ->getNewLanguage (rg2getjson.js) ->setNewLanguage (rg2ui.js) ->createEventMenu (rg2ui.js)
  ->createObjects();
Menu objects like rg2.events are not initialized at this point, initialization is done just after rg2.setLanguageOptions call

    setNewLanguage : function (lang) {
      var eventid;
      if ($("#rg2-event-ul").menu("instance") !== undefined) {
        $("#rg2-event-ul").menu("destroy");
      }
      // non-english dictionary is already installed by script that has loaded

      if (rg2.events !== undefined) {
        this.createEventMenu();

          this.createCourseMenu();
          this.createResultMenu();
        }
      }
      $("#rg2-info-panel").tabs("refresh");
      rg2.redraw(false);
